Quick Start Example

JavaScript/TypeScript 
import { createPaymentRequest, verifyPayment } from "@bnbpay/sdk";

// Create a payment request
const request = createPaymentRequest({
  chainId: 56,
  recipient: "0xMerchantAddress",
  amount: "10.00",
  currency: "USD",
  token: "0x55d398326f99059fF775485246999027B3197955", // USDT
  referenceId: "order_12345"
});

// Generate payment URI and QR code
const paymentURI = request.toURI();
const qrCode = request.toQR();
